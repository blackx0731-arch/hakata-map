// ── スポットリスト描画 ──
function renderList(filter) {
  const list = document.getElementById('spot-list');
  list.innerHTML = '';
  const filtered = spots.filter(s => filter === 'all' || s.type === filter);
  filtered.forEach(spot => {
    const cfg = typeConfig[spot.type];
    const item = document.createElement('div');
    item.className = 'spot-item';
    item.innerHTML = `
      <div class="spot-icon ${cfg.iconClass}">${cfg.icon}</div>
      <span class="spot-name">${spot.name}</span>
      <span class="spot-tag ${cfg.tagClass}">${cfg.label}</span>
      <button class="nav-btn" title="マップで表示">›</button>
    `;
    item.onclick = () => showSpot(spot);
    list.appendChild(item);
  });
}

// ── タブ切替 ──
const tabFilters = { venues: ['paid','free'], info: ['info'], toilet: ['toilet'] };
function switchTab(tab) {
  document.querySelectorAll('.tab').forEach((t,i)=>{
    t.classList.toggle('active', ['venues','info','toilet'][i] === tab);
  });
  const list = document.getElementById('spot-list');
  list.innerHTML = '';
  const types = tabFilters[tab];
  const filtered = spots.filter(s => types.includes(s.type));
  filtered.forEach(spot => {
    const cfg = typeConfig[spot.type];
    const item = document.createElement('div');
    item.className = 'spot-item';
    item.innerHTML = `
      <div class="spot-icon ${cfg.iconClass}">${cfg.icon}</div>
      <span class="spot-name">${spot.name}</span>
      <span class="spot-tag ${cfg.tagClass}">${cfg.label}</span>
      <button class="nav-btn" title="マップで表示">›</button>
    `;
    item.onclick = () => showSpot(spot);
    list.appendChild(item);
  });
}

// ── スポット選択 ──
function showSpot(spot) {
  currentSpot = spot;
  document.getElementById('popup-name').textContent = spot.name;
  document.getElementById('popup-desc').textContent = spot.desc;
  document.getElementById('info-popup').classList.add('show');
  if (map) map.panTo({ lat: spot.lat, lng: spot.lng });
}

function closePopup() {
  document.getElementById('info-popup').classList.remove('show');
}

// ── ルート表示 ──
function openRoute() {
  if (!currentSpot) return;
  const { lat, lng, name } = currentSpot;
  // スマホはGoogleマップアプリへ、PCはブラウザで開く
  const url = `https://www.google.com/maps/dir/?api=1&destination=${lat},${lng}&destination_place_id=&travelmode=walking`;
  window.open(url, '_blank');
}

// ── 現在地取得 ──
function locateMe() {
  if (!navigator.geolocation) { alert('位置情報が利用できません'); return; }
  navigator.geolocation.getCurrentPosition(pos => {
    const latlng = { lat: pos.coords.latitude, lng: pos.coords.longitude };
    if (!map) return;
    if (userMarker) userMarker.setMap(null);
    userMarker = new google.maps.Marker({
      position: latlng,
      map,
      icon: {
        path: google.maps.SymbolPath.CIRCLE,
        scale: 8,
        fillColor: '#4da6ff',
        fillOpacity: 1,
        strokeColor: '#fff',
        strokeWeight: 2,
      },
      title: '現在地',
    });
    map.panTo(latlng);
    map.setZoom(16);
  }, () => { alert('現在地を取得できませんでした。\nブラウザの位置情報を許可してください。'); });
}

// ── Google Maps 初期化 ──
function initMap() {
  // プレースホルダーを非表示にしてマップを表示
  document.getElementById('map-placeholder').style.display = 'none';
  document.getElementById('google-map').style.display = 'block';

  map = new google.maps.Map(document.getElementById('google-map'), {
    center: { lat: 33.5897, lng: 130.4138 },
    zoom: 16,
    styles: [
      { elementType:'geometry', stylers:[{color:'#1a1628'}] },
      { elementType:'labels.text.fill', stylers:[{color:'#9e8e6a'}] },
      { elementType:'labels.text.stroke', stylers:[{color:'#1a1628'}] },
      { featureType:'road', elementType:'geometry', stylers:[{color:'#2a2240'}] },
      { featureType:'road', elementType:'labels.text.fill', stylers:[{color:'#6b5f8a'}] },
      { featureType:'water', elementType:'geometry', stylers:[{color:'#0d1b2a'}] },
      { featureType:'poi.park', elementType:'geometry', stylers:[{color:'#1a2a1a'}] },
      { featureType:'transit', elementType:'geometry', stylers:[{color:'#231f35'}] },
    ],
    disableDefaultUI: false,
    zoomControl: true,
    mapTypeControl: false,
    streetViewControl: false,
    fullscreenControl: false,
  });

  // マーカーを配置
  spots.forEach(spot => {
    const cfg = typeConfig[spot.type];
    const marker = new google.maps.Marker({
      position: { lat: spot.lat, lng: spot.lng },
      map,
      title: spot.name,
      label: {
        text: cfg.icon,
        fontSize: '16px',
      },
    });
    marker.addListener('click', () => showSpot(spot));
    markers.push(marker);
  });
}
